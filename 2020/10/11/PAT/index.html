A1034

2020-10-11

逻辑

图的遍历DFS
大致思路分为两步，第一步建图，第二步DFS。
建图的数据输入是个难点，题目只给了边的信息，一是要将字符串信息转化为对应的顶点信息，二是需要判断每条边的两个字符串是否已经转化而不需要添加新的顶点。字符串的转化可以通过建立映射解决，每输入一个新的字符串就建立两个相互的映射，而转化的判断可使用map的find函数，如果返回值map.find(key) != map.end()则已经转化。注意字符串必须用string而不能用char数组。
建图完成后进行DFS，根据题意要求每个gang也就是对应图中的连通分量要有head，totalvalue，numMember的信息，所以在DFSTrave函数中每遇到没有遍历的顶点也就是一个新的连通分量就初始化这3个变量，并作为额外的三个引用参数传入DFS函数，在一个连通分量遍历完之后判断是否满足gang的要求即可。

代码

#include
#include
#include
using namespace std;
const int maxn = 2010;

map<int, string> IntToString;
map<string, int> StringToInt;
map<string, int> Gang;
int G[maxn][maxn] = { 0 }, weight[maxn] = { 0 };
bool vis[maxn] = { false };
int n, k, numPerson = 0;

void DFS(int nowVisit, int& numMember, int& totalvalue, int& head) {
numMember++;
vis[nowVisit] = true;
f (weight[nowVisit] > weight[head]) head = nowVisit;
for (int v = 0; v < numPerson; v++) {
if (G[nowVisit][v] > 0) {
totalvalue += G[nowVisit][v];
G[nowVisit][v] = G[v][nowVisit] = 0;
if (vis[v] == false) {
DFS(v, numMember, totalvalue, head);
}
}
}
}

void DFSTrave() {
for (int u = 0; u < numPerson; u++) {
if (vis[u] == false) {
int head = u, numMember = 0, totalvalue = 0;
DFS(u, numMember, totalvalue, head);
if (numMember > 2 && totalvalue > k) {
Gang[IntToString[head]] = numMember;
}
}
}
}

int change(string str) {
if (StringToInt.find(str) != StringToInt.end()) {
return StringToInt[str];
}
else {
StringToInt[str] = numPerson;
IntToString[numPerson] = str;
return numPerson++;
}
}

int main() {
string str1, str2;
int w;
cin >> n >> k;
for (int i = 0; i < n; i++) {
cin >> str1 >> str2 >> w;
int id1 = change(str1);
int id2 = change(str2);
weight[id1] += w;
weight[id2] += w;
G[id1][id2] += w;
G[id2][id1] += w;
}
DFSTrave();
cout << Gang.size() << endl;
map<string, int>::iterator it;
for (it = Gang.begin(); it != Gang.end(); it++) {
cout << it->first << “ “ << it->second << endl;
}
return 0;
}

图的遍历BFS
思路完全按着算法笔记的讲解来。
注意：
1.进行多次遍历要重置inq数组，用string.h头文件下的memste函数即可（memset使用的是按字节赋值，
建议只用来赋0和-1，0的补码全为0，-1的补码全为1）
2.bool型数组和Adj数组用的都是节点的id，不要搞混

代码

#include
#include
#include
#include
using namespace std;
const int MAXN = 1010;

struct Node {
int id;
int layer;
};
vector Adj[MAXN];
bool inq[MAXN] = { false };

int BFS(int s, int L) {
int numForward = 0;
queue q;
Node start;
start.id = s;
start.layer = 0;
q.push(start);
inq[start.id] = true;
while (!q.empty()) {
Node topNode = q.front();
q.pop();
int u = topNode.id;
for (int i = 0; i < Adj[u].size(); i++) {
Node next = Adj[u][i];
next.layer = topNode.layer + 1;
if (inq[next.id] == false && next.layer <= L) {
q.push(next);
inq[next.id] = true;
numForward++;
}
}
}
return numForward;
}

int main() {
Node user;
int n, L, numFollow, idFollow;
scanf(“%d%d”, &n, &L);
for (int i = 1; i <= n; i++) {
user.id = i;
scanf(“%d”, &numFollow);
for (int j = 0; j < numFollow; j++) {
scanf(“%d”, &idFollow);
Adj[idFollow].push_back(user);
}
}
int numQuery, s;
scanf(“%d”, &numQuery);
for (int i = 0; i < numQuery; i++) {
memset(inq, false, sizeof(inq));
scanf(“%d”, &s);
int numForward = BFS(s, L);
printf(“%d\n”, numForward);
}
return 0;
}

最短路径-Dijkstra
按照算法笔记的讲解就是一道模板题，建图+Dijkstra，没啥好讲的。
注意：
1.数组非零数据的初始化用到了头文件algorithm中的fill函数
2.注意每句代码的细节，写错一个变量名debug半小时。

代码

#include
#include
using namespace std;
const int MAXN = 510;
const int INF = 1000000000;

int n, m, st, ed, G[MAXN][MAXN], weight[MAXN];
int d[MAXN], w[MAXN], num[MAXN];
bool vis[MAXN] = { false };

void Dijkstra(int st) {
fill(d, d + MAXN, INF);
fill(w, w + MAXN, 0);
fill(num, num + MAXN, 0);
d[st] = 0;
w[st] = weight[st];
num[st] = 1;
for (int i = 0; i < n; i++) {
int u = -1, MIN = INF;
for (int j = 0; j < n; j++) {
if (vis[j] == false && d[j] < MIN) {
u = j;
MIN = d[j];
}
}
if (u == -1) return;
vis[u] = true;
for (int v = 0; v < n; v++) {
if(vis[v] == false && G[u][v] != INF){
if (G[u][v] + d[u] < d[v]) {
d[v] = G[u][v] + d[u];
w[v] = weight[v] + w[u];
num[v] = num[u];
}
else if (G[u][v] + d[u] == d[v]) {
if (w[u] + weight[v] > w[v]) {
w[v] = w[u] + weight[v];
}
num[v] += num[u];
}
}
}
}
}

int main() {
scanf(“%d%d%d%d”, &n, &m, &st, &ed);
for (int i = 0; i < n; i++) {
scanf(“%d”, &weight[i]);
}
fill(G[0], G[0] + MAXN * MAXN, INF);
int c1, c2;
for (int i = 0; i < m; i++) {
scanf(“%d%d”, &c1, &c2);
scanf(“%d”, &G[c1][c2]);
G[c2][c1] = G[c1][c2];
}
Dijkstra(st);
printf(“%d %d”, num[ed], w[ed]);
return 0;
}

逻辑

Dijkstra 或 Dijkstra + DFS
第一种做法与A1003相差不大，只是多了输出路径，添加一个pre[]数组和DFS函数按逆序输出数组就行。

代码

#include
#include
#include
using namespace std;
const int MAXN = 510;
const int INF = 1000000000;

int n, m, S, D, G[MAXN][MAXN], cost[MAXN][MAXN];
int d[MAXN], c[MAXN], pre[MAXN];
bool vis[MAXN];

void Dijkstra(int s) {
fill(d, d + MAXN, INF);
fill(c, c + MAXN, INF);
for (int i = 0; i < n; i++) {
pre[i] = i;
}
d[s] = 0;
c[s] = 0;
for (int i = 0; i < n; i++) {
int u = -1, MIN = INF;
for (int j = 0; j < n; j++) {
if (vis[j] == false && d[j] < MIN) {
u = j;
MIN = d[j];
}
}
if (u == -1) return;
vis[u] = true;
for (int v = 0; v < n; v++) {
if (vis[v] == false && G[u][v] != INF) {
if (d[u] + G[u][v] < d[v]) {
d[v] = d[u] + G[u][v];
c[v] = c[u] + cost[u][v];
pre[v] = u;
}
else if (d[u] + G[u][v] == d[v] && c[u] + cost[u][v] < c[v]) {
c[v] = c[u] + cost[u][v];
pre[v] = u;
}
}
}
}
}

void DFS(int v) {
if (v == S) {
printf(“%d “, S);
return;
}
DFS(pre[v]);
printf(“%d “, v);
return;
}

int main() {
scanf(“%d%d%d%d”, &n, &m, &S, &D);
fill(G[0], G[0] + MAXN * MAXN, INF);
fill(cost[0], cost[0] + MAXN * MAXN, INF);
int c1, c2;
for (int i = 0; i < m; i++) {
scanf(“%d%d”, &c1, &c2);
scanf(“%d%d”, &G[c1][c2], &cost[c1][c2]);
G[c2][c1] = G[c1][c2];
cost[c2][c1] = cost[c1][c2];
}
Dijkstra(S);
DFS(D);
printf(“%d %d\n”,d[D], c[D]);
return 0;
}

struct node {

A1043

2020-10-13

逻辑

建树+遍历
题目的要求是用输入数据和前序对比，并输出后序或NO。这里定义5个数组分别存储输入、两种前序和两种后序，它们的比较用for循环对比每位数实现。（vector型变量可以直接比较，定义5个vector型变量代替数组是一种优化。）
建树及遍历中唯一注意的是题目定义的镜像树的遍历其实只是调换树的左右子树遍历顺序而已。
输入数据和前序的对比结果靠定义两个个int型变量分别记录，如果for循环出现不相等的情况，则更改初始值。
最后根据判定结果进行输出，注意树与镜像树的前序是有可能相等的，也就是记录结果两个int型变量都能输出yes，此题中有两个测试点是这种情况，这时候只需要输出其中一个就行，这里用一个if-else实现。
注意：
1.每道题的正确输出只能有一种结果，多种情况符合输出的时候需要把输出语句并列起来，这里的代码用的if-else实现。
2.要注意读题，英文没读懂是硬伤。。。

代码

#include<stdio.h>
const int MAXN = 1010;

struct node {
int data;
node* lchild;
node* rchild;
}Node[MAXN];
int n;
int d[MAXN];
int pre[MAXN], mpre[MAXN], post[MAXN], mpost[MAXN];
int numpre = 0, nummpre = 0, numpost = 0, nummpost = 0;

node* newNode(int x) {
node* Node = new node;
Node->data = x;
Node->lchild = Node->rchild = NULL;
return Node;
}
void insert(node*& root, int x) {
if (root == NULL) {
root = newNode(x);
return;
}
if (x < root->data) {
insert(root->lchild, x);
}
else {
insert(root->rchild, x);
}
}

node* create(int d[], int n) {
node* root = NULL;
for (int i = 0; i < n; i++) {
insert(root, d[i]);
}
return root;
}

void preorder(node* root) {
if (root == NULL) return;
pre[numpre++] = root->data;
preorder(root->lchild);
preorder(root->rchild);
}
void mpreorder(node* root) {
if (root == NULL) return;
mpre[nummpre++] = root->data;
mpreorder(root->rchild);
mpreorder(root->lchild);
}
void postorder(node* root) {
if (root == NULL) return;
postorder(root->lchild);
postorder(root->rchild);
post[numpost++] = root->data;
}
void mpostorder(node* root) {
if (root == NULL) return;
mpostorder(root->rchild);
mpostorder(root->lchild);
mpost[nummpost++] = root->data;
}
int main() {
scanf(“%d”, &n);
for (int i = 0; i < n; i++) {
scanf(“%d”, &d[i]);
}
node* root = create(d, n);
preorder(root);
mpreorder(root);
postorder(root);
mpostorder(root);
int p = 1, m = 1;
for (int i = 0; i < n; i++) {
if (pre[i] != d[i]) {
p = 0;
}
if (mpre[i] != d[i]) {
m = 0;
}
}
int num = 0;
if (p == 1) {
printf(“YES\n”);
for (int i = 0; i < n; i++) {
printf(“%d”, post[i]);
num++;
if (num < n) {
printf(“ “);
}
}
}
else if(m == 1) {
printf(“YES\n”);
for (int i = 0; i < n; i++) {
printf(“%d”, mpost[i]);
num++;
if (num < n) {
printf(“ “);
}
}
}
else if (p == 0 && m == 0) {
printf(“NO”);
}
return 0;
}

if (sum > s) return;

逻辑

建树+层次遍历
在层次遍历的基础上如何判断一层结束是个问题，这里在节点定义一个layer记录层数，再在遍历过程中定义一个nowlayer记录当前层数，当layer大于nowlayer时则说明一层结束，此时应该输出叶子节点个数，末尾节点是个特殊情况因为此时layer不能大于nowlayer，所以对于特殊情况新增一个变量记录已访问的节点数作为对末尾节点判断。
注意：
1.建树过程中layer的赋值应该在树形成之后，不然按这里的代码方式进行赋值会产生遗漏。我最开始在读数据时就赋值layer始终有一个测试点未通过，把赋值语句移到层序遍历中就能通过了。

代码

#include
#include
#include
using namespace std;
const int maxn = 110;

struct node {
int layer;
vector child;
}Node[maxn];
int n, m;

void LayerTrave(int root) {
queue q;
q.push(root);
int nowlayer = 1, ans = 0, num = 0;
while (!q.empty()) {
int now = q.front();
q.pop();
num++;
if (nowlayer == Node[now].layer) {
if (Node[now].child.size() == 0) {
ans++;
}
}
else {
printf(“%d “, ans);
if (Node[now].child.size() == 0) {
ans = 1;
}
else {
ans = 0;
}
nowlayer++;
}
if (num == n) {
printf(“%d\n”, ans);
return;
}
for (int i = 0; i < Node[now].child.size(); i++) {
int child = Node[now].child[i];
q.push(Node[now].child[i]);
Node[child].layer = Node[now].layer + 1;
}
}
}

int main() {
scanf(“%d%d”, &n, &m);
Node[1].layer = 1;
int id, num, child;
for (int i = 0; i < m; i++) {
scanf(“%d%d”, &id, &num);
for (int j = 0; j < num; j++) {
scanf(“%d”, &child);
Node[id].child.push_back(child);
}
}
LayerTrave(1);
return 0;
}

逻辑

映射+排序

逻辑

并查树+DFS
先用并查树判断输入数据是否是一棵树，再DFS遍历树得到树高并选出最大值，多个值用set存储
注意：
1.isRoot的用法。bool型变量为true时对应值1.
2.set型变量的遍历只能用it。

代码

#include
#include
#include
#include
using namespace std;
const int N = 100010;

vector G[N];
set deepest;
bool isRoot[N];
bool vis[N];
int n;
int maxheight = 1, treeheight = 1;
int father[N];

int findfather(int x) {
int a = x;
while (x != father[x]) {
x = father[x];
}
while (a != father[a]) {
int z = a;
a = father[a];
father[z] = x;
}
return x;
}

void Union(int a, int b) {
int faA = findfather(a);
int faB = findfather(b);
if (faA != faB) {
father[faA] = faB;
}
}

int calBlock(int n) {
int Block = 0;
memset(isRoot, false, sizeof(isRoot));
for (int i = 1; i <= n; i++) {
isRoot[findfather(i)] = true;
}
for (int i = 1; i <= n; i++) {
Block += isRoot[i];
}
return Block;
}

void TreeDFS(int v, int height) {
vis[v] = true;
if (height > treeheight) {
treeheight = height;
}
for (int i = 0; i < G[v].size(); i++) {
if (vis[G[v][i]] == false) {
TreeDFS(G[v][i], height + 1);
}
}
}

int main() {
cin >> n;
for (int i = 1; i <= n; i++) {
father[i] = i;
}
for (int i = 0; i < n - 1; i++) {
int u, v;
cin >> u >> v;
G[u].push_back(v);
G[v].push_back(u);
Union(u, v);
}
if (calBlock(n) != 1) {
cout << “Error: “ << calBlock(n) << “ components”;
return 0;
}
for (int i = 1; i <= n; i++) {
memset(vis, false, sizeof(vis));
TreeDFS(i, 1);
if (treeheight > maxheight) {
maxheight = treeheight;
deepest.clear();
deepest.insert(i);
}
else if (maxheight == treeheight) {
deepest.insert(i);
}
treeheight = 1;
}
for (set::iterator it = deepest.begin(); it != deepest.end(); it++) {
cout << *it << endl;
}
return 0;
}

逻辑

Dijkstra+DFS
根据题意有第二标尺和第三标尺，且不能只是用Dijkstra来解决，因为它们在路径上的传递不满足最优子结构，值会受后面顶点的影响，所有在Dijkstra后用DFS。
注意：
1.数组一般情况默认初值为0。
2.题目给了n个数，数的下标从1开始，Dijkstra应该进行n+1轮遍历n+1个顶点。

代码

#include
#include
#include
using namespace std;
const int maxn = 510;
const int INF = 1000000000;

int G[maxn][maxn], weight[maxn];
int d[maxn];
bool vis[maxn];
int c, n, sp, m;
vector pre[maxn];
vector path, temppath;
int minneed = INF, minremain = INF;

void Dijkstra(int s) {
fill(d, d + maxn, INF);
d[s] = 0;
for (int i = 0; i <= n; i++) {
int u = -1, min = INF;
for (int j = 0; j <= n; j++) {
if (vis[j] == false && d[j] < min) {
u = j;
min = d[j];
}
}
if (u == -1) return;
vis[u] = true;
for (int v = 0; v <= n; v++) {
if (vis[v] == false && G[u][v] != INF) {
if (d[u] + G[u][v] < d[v]) {
d[v] = d[u] + G[u][v];
pre[v].clear();
pre[v].push_back(u);
}
else if (d[u] + G[u][v] == d[v]) {
pre[v].push_back(u);
}
}
}
}
}

void DFS(int v) {
if (v == 0) {
temppath.push_back(v);
int need = 0, remain = 0;
for (int i = temppath.size() - 1; i >= 0; i–) {
int id = temppath[i];
if (weight[id] > 0) {
remain += weight[id];
}
else {
if (remain > abs(weight[id])) {
remain -= abs(weight[id]);
}
else {
need += abs(weight[id]) - remain;
remain = 0;
}
}
}
if (need < minneed) {
minneed = need;
minremain = remain;
path = temppath;
}
else if (need == minneed && remain < minremain) {
minremain = remain;
path = temppath;
}
temppath.pop_back();
return;
}
temppath.push_back(v);
for (int i = 0; i < pre[v].size(); i++) {
DFS(pre[v][i]);
}
temppath.pop_back();
}

int main() {
cin >> c >> n >> sp >> m;
for (int i = 1; i <= n; i++) {
cin >> weight[i];
weight[i] -= c / 2;
}
fill(G[0], G[0] + maxn * maxn, INF);
for (int i = 0; i < m; i++) {
int u, v;
cin >> u >> v;
cin >> G[u][v];
G[v][u] = G[u][v];
}
Dijkstra(0);
DFS(sp);
cout << minneed <<” “;
for (int i = path.size() - 1; i >= 0; i–) {
cout << path[i];
if (i > 0) cout << “->”;
}
cout << “ “ << minremain;
return 0;
}
