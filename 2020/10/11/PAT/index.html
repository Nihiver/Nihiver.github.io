A1034

2020-10-11

逻辑

图的遍历DFS
大致思路分为两步，第一步建图，第二步DFS。
建图的数据输入是个难点，题目只给了边的信息，一是要将字符串信息转化为对应的顶点信息，二是需要判断每条边的两个字符串是否已经转化而不需要添加新的顶点。字符串的转化可以通过建立映射解决，每输入一个新的字符串就建立两个相互的映射，而转化的判断可使用map的find函数，如果返回值map.find(key) != map.end()则已经转化。注意字符串必须用string而不能用char数组。
建图完成后进行DFS，根据题意要求每个gang也就是对应图中的连通分量要有head，totalvalue，numMember的信息，所以在DFSTrave函数中每遇到没有遍历的顶点也就是一个新的连通分量就初始化这3个变量，并作为额外的三个引用参数传入DFS函数，在一个连通分量遍历完之后判断是否满足gang的要求即可。

代码

#include
#include
#include
using namespace std;
const int maxn = 2010;

map<int, string> IntToString;
map<string, int> StringToInt;
map<string, int> Gang;
int G[maxn][maxn] = { 0 }, weight[maxn] = { 0 };
bool vis[maxn] = { false };
int n, k, numPerson = 0;

void DFS(int nowVisit, int& numMember, int& totalvalue, int& head) {
numMember++;
vis[nowVisit] = true;
f (weight[nowVisit] > weight[head]) head = nowVisit;
for (int v = 0; v < numPerson; v++) {
if (G[nowVisit][v] > 0) {
totalvalue += G[nowVisit][v];
G[nowVisit][v] = G[v][nowVisit] = 0;
if (vis[v] == false) {
DFS(v, numMember, totalvalue, head);
}
}
}
}

void DFSTrave() {
for (int u = 0; u < numPerson; u++) {
if (vis[u] == false) {
int head = u, numMember = 0, totalvalue = 0;
DFS(u, numMember, totalvalue, head);
if (numMember > 2 && totalvalue > k) {
Gang[IntToString[head]] = numMember;
}
}
}
}

int change(string str) {
if (StringToInt.find(str) != StringToInt.end()) {
return StringToInt[str];
}
else {
StringToInt[str] = numPerson;
IntToString[numPerson] = str;
return numPerson++;
}
}

int main() {
string str1, str2;
int w;
cin >> n >> k;
for (int i = 0; i < n; i++) {
cin >> str1 >> str2 >> w;
int id1 = change(str1);
int id2 = change(str2);
weight[id1] += w;
weight[id2] += w;
G[id1][id2] += w;
G[id2][id1] += w;
}
DFSTrave();
cout << Gang.size() << endl;
map<string, int>::iterator it;
for (it = Gang.begin(); it != Gang.end(); it++) {
cout << it->first << “ “ << it->second << endl;
}
return 0;
}

A1076

2020-10-11

逻辑

图的遍历BFS
思路完全按着算法笔记的讲解来。
注意：
1.进行多次遍历要重置inq数组，用string.h头文件下的memste函数即可（memset使用的是按字节赋值，
建议只用来赋0和-1，0的补码全为0，-1的补码全为1）
2.bool型数组和Adj数组用的都是节点的id，不要搞混

代码

#include
#include
#include
#include
using namespace std;
const int MAXN = 1010;

struct Node {
int id;
int layer;
};
vector Adj[MAXN];
bool inq[MAXN] = { false };

int BFS(int s, int L) {
int numForward = 0;
queue q;
Node start;
start.id = s;
start.layer = 0;
q.push(start);
inq[start.id] = true;
while (!q.empty()) {
Node topNode = q.front();
q.pop();
int u = topNode.id;
for (int i = 0; i < Adj[u].size(); i++) {
Node next = Adj[u][i];
next.layer = topNode.layer + 1;
if (inq[next.id] == false && next.layer <= L) {
q.push(next);
inq[next.id] = true;
numForward++;
}
}
}
return numForward;
}

int main() {
Node user;
int n, L, numFollow, idFollow;
scanf(“%d%d”, &n, &L);
for (int i = 1; i <= n; i++) {
user.id = i;
scanf(“%d”, &numFollow);
for (int j = 0; j < numFollow; j++) {
scanf(“%d”, &idFollow);
Adj[idFollow].push_back(user);
}
}
int numQuery, s;
scanf(“%d”, &numQuery);
for (int i = 0; i < numQuery; i++) {
memset(inq, false, sizeof(inq));
scanf(“%d”, &s);
int numForward = BFS(s, L);
printf(“%d\n”, numForward);
}
return 0;
}

A1003

2020-10-11

逻辑

最短路径-Dijkstra
按照算法笔记的讲解就是一道模板题，建图+Dijkstra，没啥好讲的。
注意：
1.数组非零数据的初始化用到了头文件algorithm中的fill函数
2.注意每句代码的细节，写错一个变量名debug半小时。

代码

#include
#include
using namespace std;
const int MAXN = 510;
const int INF = 1000000000;

int n, m, st, ed, G[MAXN][MAXN], weight[MAXN];
int d[MAXN], w[MAXN], num[MAXN];
bool vis[MAXN] = { false };

void Dijkstra(int st) {
fill(d, d + MAXN, INF);
fill(w, w + MAXN, 0);
fill(num, num + MAXN, 0);
d[st] = 0;
w[st] = weight[st];
num[st] = 1;
for (int i = 0; i < n; i++) {
int u = -1, MIN = INF;
for (int j = 0; j < n; j++) {
if (vis[j] == false && d[j] < MIN) {
u = j;
MIN = d[j];
}
}
if (u == -1) return;
vis[u] = true;
for (int v = 0; v < n; v++) {
if(vis[v] == false && G[u][v] != INF){
if (G[u][v] + d[u] < d[v]) {
d[v] = G[u][v] + d[u];
w[v] = weight[v] + w[u];
num[v] = num[u];
}
else if (G[u][v] + d[u] == d[v]) {
if (w[u] + weight[v] > w[v]) {
w[v] = w[u] + weight[v];
}
num[v] += num[u];
}
}
}
}
}

int main() {
scanf(“%d%d%d%d”, &n, &m, &st, &ed);
for (int i = 0; i < n; i++) {
scanf(“%d”, &weight[i]);
}
fill(G[0], G[0] + MAXN * MAXN, INF);
int c1, c2;
for (int i = 0; i < m; i++) {
scanf(“%d%d”, &c1, &c2);
scanf(“%d”, &G[c1][c2]);
G[c2][c1] = G[c1][c2];
}
Dijkstra(st);
printf(“%d %d”, num[ed], w[ed]);
return 0;
}

A1030

逻辑

Dijkstra 或 Dijkstra + DFS
第一种做法与A1003相差不大，只是多了输出路径，添加一个pre[]数组和DFS函数按逆序输出数组就行。
第二种做法明天再写。。。

代码

#include
#include
#include
using namespace std;
const int MAXN = 510;
const int INF = 1000000000;

int n, m, S, D, G[MAXN][MAXN], cost[MAXN][MAXN];
int d[MAXN], c[MAXN], pre[MAXN];
bool vis[MAXN];

void Dijkstra(int s) {
fill(d, d + MAXN, INF);
fill(c, c + MAXN, INF);
for (int i = 0; i < n; i++) {
pre[i] = i;
}
d[s] = 0;
c[s] = 0;
for (int i = 0; i < n; i++) {
int u = -1, MIN = INF;
for (int j = 0; j < n; j++) {
if (vis[j] == false && d[j] < MIN) {
u = j;
MIN = d[j];
}
}
if (u == -1) return;
vis[u] = true;
for (int v = 0; v < n; v++) {
if (vis[v] == false && G[u][v] != INF) {
if (d[u] + G[u][v] < d[v]) {
d[v] = d[u] + G[u][v];
c[v] = c[u] + cost[u][v];
pre[v] = u;
}
else if (d[u] + G[u][v] == d[v] && c[u] + cost[u][v] < c[v]) {
c[v] = c[u] + cost[u][v];
pre[v] = u;
}
}
}
}
}

void DFS(int v) {
if (v == S) {
printf(“%d “, S);
return;
}
DFS(pre[v]);
printf(“%d “, v);
return;
}

int main() {
scanf(“%d%d%d%d”, &n, &m, &S, &D);
fill(G[0], G[0] + MAXN * MAXN, INF);
fill(cost[0], cost[0] + MAXN * MAXN, INF);
int c1, c2;
for (int i = 0; i < m; i++) {
scanf(“%d%d”, &c1, &c2);
scanf(“%d%d”, &G[c1][c2], &cost[c1][c2]);
G[c2][c1] = G[c1][c2];
cost[c2][c1] = cost[c1][c2];
}
Dijkstra(S);
DFS(D);
printf(“%d %d\n”,d[D], c[D]);
return 0;
}

A1020

2020-10-12

逻辑

重建二叉树+层次遍历
注意：
1.又写错变量名——改代码半小时
2.输出格式有要求，应该是在最后一个输出之前每输出一个数就应该输出一个空格，可用num计数，每自增就判断是否要输出空格。

代码

nclude
#include
using namespace std;
const int MAXN = 40;

struct node {
int data;
node* lchild;
node* rchild;
};

int n;
int PostOrder[MAXN], InOrder[MAXN];

node* create(int postL, int postR, int inL, int inR) {
if (postL > postR) {
return NULL;
}
node* root = new node;
root->data = PostOrder[postR];
int k;
for (k = inL; k <= inR; k++) {
if (InOrder[k] == PostOrder[postR]) {
break;
}
}
int numLeft = k - inL;
root->lchild = create(postL, postL + numLeft - 1, inL, k - 1);
root->rchild = create(postL + numLeft, postR - 1, k + 1, inR);
return root;
}

void LevelOrder(node* root) {
queue<node*> q;
q.push(root);
int num = 0;
while (!q.empty()) {
node* now = q.front();
q.pop();
printf(“%d”, now->data);
num++;
if (now->lchild != NULL) {
q.push(now->lchild);
}
if (now->rchild != NULL) {
q.push(now->rchild);
}
if (num < n) {
printf(“ “);
}
}
}

int main() {
scanf(“%d”, &n);
for (int i = 0; i < n; i++) {
scanf(“%d”, &PostOrder[i]);
}
for (int j = 0; j < n; j++) {
scanf(“%d”, &InOrder[j]);
}
node* root = create(0, n - 1, 0, n - 1);
LevelOrder(root);
return 0;
}
