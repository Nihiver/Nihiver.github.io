逻辑

给出一组学生的准考证号和成绩，准考证号包含了等级 (乙甲顶)，考场号，日期，和个人编号信息，并有三种查询方式
查询一：给出考试等级，找出该等级的考生，按照成绩降序，准考证升序排序
查询二：给出考场号，统计该考场的考生数量和总得分
查询三：给出考试日期，查询改日期下所有考场的考试人数，按照人数降序，考场号升序排序

分析：
1.string 的输入必须用 cin，但用 cout 输出会超时，应使用 c_str（）类型转化之后用 printf 输出。
2.map 建立映射但需另外排序，可以用 for (auto it : map) 存入 vector 容器中，再排序输出。
3. 排序传参用引用传参会更快。
4.unorderer_map 比 map 更快。
5. 字符串的输入用 string 处理，既可以直接判断是否相等，还可以用 substr 直接取子串。

—-string str;
—-int value;

—-return a.value != b.value ? a.value > b.value : a.str < b.str;

—-cin >> n >> m;
—-vector< node> v (n);
—-for (int i = 0; i < n; i++) {
——–cin >> v [i].str >> v [i].value;
—-}
—-int num;
—-string s;
—-for (int i = 1; i <= m; i++) {
——–cin >> num >> s;
——–printf (“Case % d: % d % s\n”, i, num, s.c_str ());
——–vector< node> ans;
——–if (num == 1) {
————for (int j = 0; j < n; j++) {
—————-if (v [j].str [0] == s [0]) {
——————–ans.push_back (v [j]);
—————-}
————}
————sort (ans.begin (), ans.end (), cmp);
————for (int j = 0; j < ans.size (); j++) {
—————-printf (“% s % d\n”, ans [j].str.c_str (), ans [j].value);
————}
————if (ans.size () == 0) cout << “NA” << endl;
——–}
——–else if (num == 2) {
————int number = 0, sum = 0;
————for (int j = 0; j < n; j++) {
—————-if (v [j].str.substr (1, 3) == s) {
——————–number++;
——————–sum += v [j].value;
—————-}
————}
————if (number == 0) {
—————-cout << “NA” << endl;
————}
————else {
—————-cout << number << “ “ << sum << endl;
————}
——–}
——–else if (num == 3) {
————unordered_map<string, int> mp;
————for (int j = 0; j < n; j++) {
—————-if (v [j].str.substr (4, 6) == s) {
——————–mp [v [j].str.substr (1, 3)]++;
—————-}
————}
————for (auto it : mp) ans.push_back ({ it.first, it.second });
————sort (ans.begin (), ans.end (), cmp);
————for (int j = 0; j < ans.size (); j++) {
—————-printf (“% s % d\n”, ans [j].str.c_str (), ans [j].value);
————}

—-}

给一个图，以点权代表颜色，如果每条边的两个顶点颜色都不相同，那这个图就是一个 k-coloring 图。
分析：

  1. 可以用 set 存储点权，那么 set 的大小就是颜色的多少。

—-for (int u = 0; u < n; u++) {
——–for (int j = 0; j < Adj [u].size (); j++) {
————int v = Adj [u][j];

——–}
—-}
—-return true;

int main () {
—-scanf (“% d% d”, &n, &m);
—-for (int i = 0; i < m; i++) {
——–int u, v;
——–scanf (“% d% d”, &u, &v);
——–Adj [u].push_back (v);
——–Adj [v].push_back (u);
—-}
—-int k;
—-scanf (“% d”, &k);
—-for (int i = 0; i < k; i++) {
——–set< int> s;
——–for (int j = 0; j < n; j++) {
————scanf (“% d”, &col [j]);
————s.insert (col [j]);
——–}
——–if (isKcoloring ()) {
————printf (“% d-coloring\n”, s.size ());
——–}
——–else {
————printf (“No\n”);
——–}
—-}
—-return 0;
}

给出一个完全二叉树，输出从根节点到右节点的路径 (先右子树后左子树)，并判断是不是堆。
路径的逻辑：DFS 树，每到一个节点就将该节点存入路径，如果它是叶子节点则输出路径，否则继续遍历。
堆的判断就比较每个父节点和叶子节点，用两个变量记录比较结果，最终可得到三种结果。

代码

#include< cstdio>
#include< vector>

—-if (index > n) return;
—-path.push_back (tree [index]);
—-if (index * 2 > n) {
——–for (int i = 0; i < path.size (); i++) {
————printf (“% d% s”, path [i], i != path.size () - 1 ? “” :”\n”);
——–}
——–path.pop_back ();
——–return;
—-}
—-DFS (index * 2 + 1);
—-DFS (index * 2);
—-path.pop_back ();
}

int main () {
—-scanf (“% d”, &n);
—-for (int i = 1; i <= n; i++) {
——–scanf (“% d”, &tree [i]);
—-}
—-DFS (1);
—-int isMin = 1, isMax = 1;
—-for (int i = 2; i <= n; i++) {
——–if (tree [i / 2] > tree [i]) isMin = 0;
——–if (tree [i / 2] < tree [i]) isMax = 0;
—-}
—-if (isMin == 1)
——–printf (“Min Heap”);
—-else
——–printf (“% s”, isMax == 1 ? “Max Heap” : “Not Heap”);
—-return 0;
}

N 个人里面有两个狼人，其他全是好人，有一个狼人和一个好人说谎，找出这两个狼人。
根据题意要输出下标最小的两个狼人，下标要最小肯定会用到从最小值到最大值的 for 循环，假定循环的当前参数值就是狼人的下标。单独设一个 a 数组记录每个人的身份，狼人的值为 - 1，好人的值为 1。v 数组记录每个人的发言，如果 v*a<0, 则说明他在说谎，说谎的人存入 lie 数组。如果遍历结束 lie 中有两个人且一个是狼人一个是好人，则得到一组结果。

代码

#include< vector>

int main () {
—-scanf (“% d”, &n);
—-vector< int> v (n + 1);
—-for (int i = 1; i <= n; i++) scanf (“% d”, &v [i]);
—-for (int i = 1; i < n; i++) {
——–for (int j = i + 1; j <= n; j++) {
————vector< int> lie, a (n + 1, 1);
————a [i] = a [j] = -1;
————for (int k = 1; k <= n; k++) {
—————-if (v [k] * a [abs (v [k])] < 0) {
——————–lie.push_back (k);
—————-}
————}
————if (lie.size () == 2 && a [lie [0]] + a [lie [1]] == 0) {
—————-printf (“% d % d\n”, i, j);
—————-return 0;
————}
——–}
—-}
—-printf (“No Solution\n”);

给一个清单，里面的每一对物品不能放在一起，再给出几个清单，判断里面的物品能不能放在一起。
分析：

  1. 一个物品不能共存的物品可以用一个 vector 数组存储，可建立一个 map< int, vector< int>> 的映射实现。
  2. 对清单中的每个物品都应该遍历它的不共存数组与当前清单中的每个物品对比，看是否是同一个即值相等，会有 3 个循环：清单、不共存数组、清单。这里可用空间换时间的方式，设一个 a 数组包含全部范围的物品，在清单里的物品值为 1，这样就可以省去第三个循环，直接判断 a 的值是否为 1 就行。

#include< vector>
#include< map>
using namespace std;
int main () {
—-int n, m;
—-scanf (“% d% d”, &n, &m);
—-map< int, vector< int> > p;
—-for (int i = 0; i < n; i++) {
——–int u, v;
——–scanf (“% d% d”, &u, &v);
——–p [u].push_back (v);
——–p [v].push_back (u);
—-}
—-for (int i = 0; i < m; i++) {
——–int num , flag = 1, a [100000] = { 0 };
——–scanf (“% d”, &num);
——–vector< int> v (num);
——–for (int j = 0; j < num; j++) {
————scanf (“% d”, &v [j]);
————a [v [j]] = 1;
——–}
——–for (int j = 0; j < num; j++) {
————for (int k = 0; k < p [v [j]].size (); k++) {
—————-if (a [p [v [j]][k]] == 1) flag = 0;
————}
——–}
——–printf (“% s\n”, flag ? “Yes” : “No”);
—-}
—-return 0;
}

给出一个图，然后给出几条路径，判断是不是环路图（还要区分是不是简单的）。如果没遍历所有顶点或者首尾不一或者路不通都不是环路图。
分析：

  1. 路过顶点用 set 存储，size 等于顶点数则经过所有顶点。
  2. 距离之和大于 INF 则表示不通

——–int u, v;
——–scanf (“% d% d”, &u, &v);
——–scanf (“% d”, &G [u][v]);
——–G [v][u] = G [u][v];
—-}
—-int k, num;
—-scanf (“% d”, &k);
—-int shortest = INF, d = 0;

——–set nodes;

——–for (int j = 0; j < num; j++) {

————nodes.insert (path [j]);
——–}
——–printf (“Path % d:”, i);
——–if (totaldist >= INF) {
————printf (“NA (Not a TS cycle)\n”);
——–}
——–else {
————printf (“% d”, totaldist);

—————-if (path.size () == n + 1) {
——————–printf (“(TS simple cycle)\n”);
—————-}

——————–printf (“(TS cycle)\n”);

——————–shortest = totaldist;
——————–d = i;
—————-}
————}
————else {
—————-printf (“(Not a TS cycle)\n”);

——–}
—-}

}

逻辑

柳神！
给出中序序列和先序序列，再给出两个点，求这两个点的最近公共祖先～
不用建树～已知某个树的根结点，若 a 和 b 在根结点的左边，则 a 和 b 的最近公共祖先在当前子树根结点的左子树寻找，如果 a 和 b 在当前子树根结点的两边，在当前子树的根结点就是 a 和 b 的最近公共祖先，如果 a 和 b 在当前子树根结点的右边，则 a 和 b 的最近公共祖先就在当前子树的右子树寻找。
分析：
1.vector 的 resize（）挺好用
2. 遇到查找某个数据是否存在的题目统统用 map
3. 根节点通过 pre 数组给出，每递归一次下标加一。每个点在 in 数组的位置用映射实现（没有初始化的话值为 0）

代码

—-}
—-else if (uin > inRoot && vin > inRoot) {
——–fac (inRoot + 1, inR, preRoot + 1 + (inRoot - inL), u, v);
—-}
—-else if ((uin < inRoot && vin > inRoot) || (uin > inRoot && vin < inRoot)) {
——–printf (“LCA of % d and % d is % d.\n”, u, v, in [inRoot]);
—-}
—-else if (uin == inRoot) {
——–printf (“% d is an ancestor of % d.\n”, u, v);
—-}
—-else if (vin == inRoot) {
——–printf (“% d is an ancestor of % d.\n”, v, u);
—-}
}

int main () {
—-int m, n;
—-scanf (“% d% d”, &m, &n);
—-in.resize (n + 1), pre.resize (n + 1);
—-for (int i = 1; i <= n; i++) {
——–scanf (“% d”, &in [i]);
——–pos [in [i]] = i;
—-}
—-for (int i = 1; i <= n; i++) scanf (“% d”, &pre [i]);
—-int u, v;
—-for (int i = 0; i < m; i++) {
——–scanf (“% d% d”, &u, &v);
——–if (pos [u] == 0 && pos [v] == 0) {
————printf (“ERROR: % d and % d are not found.\n”, u, v);
——–}
——–else if (pos [u] == 0 || pos [v] == 0) {
————printf (“ERROR: % d is not found.\n”, pos [u] == 0 ? u : v);
——–}
——–else {
————fac (1, n, 1, u, v);
——–}
—-}
—-return 0;
